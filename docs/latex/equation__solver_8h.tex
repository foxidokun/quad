\begin{DoxyNote}{Note}
Overflow checks\+:
\begin{DoxyEnumerate}
\item b$^\wedge$2 \texorpdfstring{$<$}{<} DOUBLE\+\_\+\+MAX

\end{DoxyEnumerate}
\end{DoxyNote}
